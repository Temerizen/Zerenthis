from __future__ import annotations

import json
import time
from pathlib import Path
from typing import Any, Dict, List

BASE = Path("backend/data")
GOALS_PATH = BASE / "generated_goals.json"
ACTIVE_GOAL_PATH = BASE / "active_goal.json"
MISSION_PATH = BASE / "active_mission.json"
INITIATIVE_PATH = BASE / "initiative_goals.json"


def _load_json(path: Path, default: Any):
    try:
        if path.exists():
            return json.loads(path.read_text(encoding="utf-8"))
    except Exception:
        pass
    return default


def _safe_write(path: Path, data: Any):
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps(data, indent=2, ensure_ascii=False), encoding="utf-8")


def _score_goal(goal: Dict[str, Any]) -> float:
    priority = float(goal.get("priority", 0.0) or 0.0)
    goal_type = str(goal.get("type", "")).strip().lower()

    type_bonus = {
        "stabilization": 0.12,
        "reflection_response": 0.08,
        "exploration": 0.05,
        "optimization": 0.04,
        "companion": 0.06,
    }.get(goal_type, 0.0)

    return round(priority + type_bonus, 4)


def select_active_goal() -> Dict[str, Any]:
    payload = _load_json(GOALS_PATH, {})
    goals = payload.get("goals", []) if isinstance(payload, dict) else []
    initiative_payload = _load_json(INITIATIVE_PATH, {})
    initiative_goals = initiative_payload.get("initiative_goals", []) if isinstance(initiative_payload, dict) else []
    if isinstance(initiative_goals, list) and initiative_goals:
        goals = list(goals) + list(initiative_goals)
    mission_payload = _load_json(MISSION_PATH, {})
    mission = mission_payload.get("mission") if isinstance(mission_payload, dict) else {}
    blocked_tasks = mission.get("blocked_tasks", []) if isinstance(mission, dict) else []

    if not isinstance(goals, list) or not goals:
        result = {
            "status": "no_goals",
            "selected_at": time.time(),
            "active_goal": None,
        }
        _safe_write(ACTIVE_GOAL_PATH, result)
        return result

    scored: List[Dict[str, Any]] = []
    for goal in goals:
        if not isinstance(goal, dict):
            continue
        enriched = dict(goal)
        enriched["_score"] = _score_goal(goal)

        goal_id = str(enriched.get("id", ""))
        blocked_hit = False
        for blocked in blocked_tasks:
            if blocked and blocked in goal_id:
                blocked_hit = True
                break

        if blocked_hit:
            enriched["_score"] = round(float(enriched["_score"]) - 0.5, 4)
            enriched["_blocked_penalty"] = True

        scored.append(enriched)

    scored.sort(key=lambda x: x.get("_score", 0.0), reverse=True)
    best = scored[0]

    result = {
        "status": "ok",
        "selected_at": time.time(),
        "active_goal": best,
        "ranked_goals": scored,
    }
    _safe_write(ACTIVE_GOAL_PATH, result)
    return result

import time
from backend.app.cognition.thought_engine import generate_thought
from backend.app.cognition.identity_engine import update_identity
from backend.app.cognition.reflection_engine import run_reflection
from backend.app.cognition.goal_generator import run_goal_generation
from backend.app.cognition.goal_arbiter import select_active_goal
from backend.app.cognition.goal_executor import execute_active_goal

from backend.app.cognition.initiative_engine import build_initiative_goals
import threading
from backend.app.execution.runner import execute_all

RUNNING = False
THREAD = None
STATE = {
    "running": False,
    "interval": 2.0,
    "cycle_count": 0,
    "last_cycle_at": None,
    "last_error": None
}

def loop(interval=2.0):
    global RUNNING
    STATE["running"] = True
    STATE["interval"] = interval
    print("[AUTONOMY] Loop started")

    while RUNNING:
        try:
            results = execute_all(limit=5)
            STATE["cycle_count"] += 1
            STATE["last_cycle_at"] = time.time()

            try:
                generate_thought(
                    current_goal="autonomy_cycle",
                    chosen_task=results[0].get("task", "unknown") if results else "no_task",
                    result="success",
                    reasoning="Completed autonomy execution cycle and recorded internal state.",
                    reflection="Cycle completed; cognition layer updated.",
                    extra={"results": results, "cycle_count": STATE["cycle_count"]}
                )
                update_identity(
                    current_goal="autonomy_cycle",
                    chosen_task=results[0].get("task", "unknown") if results else "no_task",
                    result="success",
                    goal_aligned=True
                )
                if STATE["cycle_count"] % 3 == 0:
                    run_reflection(10)

                if STATE["cycle_count"] % 4 == 0:
                    run_goal_generation()
                    build_initiative_goals()
                    select_active_goal()
                    build_active_plan()
                    build_or_update_mission()
                    resolve_conflicts()
                    execute_active_goal()
                    advance_active_plan()
                    record_execution()
                    apply_feedback()
                    advance_mission_progress()
            except Exception as cognition_error:
                print(f"[AUTONOMY][COGNITION] {cognition_error}")
            STATE["last_error"] = None
            print(f"[AUTONOMY] cycle complete: {len(results)} tasks")
        except Exception as e:
            STATE["last_error"] = str(e)
            print(f"[AUTONOMY] error: {e}")

        time.sleep(interval)

    STATE["running"] = False
    print("[AUTONOMY] Loop stopped")

def start_loop(interval=2.0):
    global RUNNING, THREAD

    if RUNNING:
        return {"status": "already_running", "state": STATE}

    RUNNING = True
    THREAD = threading.Thread(target=loop, args=(interval,), daemon=True)
    THREAD.start()
    return {"status": "started", "state": STATE}

def stop_loop():
    global RUNNING
    RUNNING = False
    return {"status": "stopping", "state": STATE}

def get_status():
    return {
        "status": "ok",
        "state": STATE
    }

from __future__ import annotations

import json
import time
from pathlib import Path
from typing import Any, Dict, List

BASE = Path("backend/data")
IDENTITY_PATH = BASE / "identity_state.json"
MISSION_PATH = BASE / "active_mission.json"
GOALS_PATH = BASE / "generated_goals.json"
INITIATIVE_PATH = BASE / "initiative_goals.json"
EXECUTION_MEMORY_PATH = BASE / "execution_memory.json"


def _load_json(path: Path, default: Any):
    try:
        if path.exists():
            return json.loads(path.read_text(encoding="utf-8"))
    except Exception:
        pass
    return default


def _save_json(path: Path, data: Any):
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps(data, indent=2, ensure_ascii=False), encoding="utf-8")


def _recent_failures(limit: int = 20) -> int:
    memory = _load_json(EXECUTION_MEMORY_PATH, {"events": []})
    events = memory.get("events", []) if isinstance(memory, dict) else []
    failures = 0
    for event in events[-limit:]:
        if isinstance(event, dict) and event.get("outcome") == "failure":
            failures += 1
    return failures


def build_initiative_goals() -> Dict[str, Any]:
    identity = _load_json(IDENTITY_PATH, {})
    mission_payload = _load_json(MISSION_PATH, {})
    mission = mission_payload.get("mission") if isinstance(mission_payload, dict) else {}
    generated = _load_json(GOALS_PATH, {})
    existing_goals = generated.get("goals", []) if isinstance(generated, dict) else []

    traits = identity.get("traits", {}) if isinstance(identity, dict) else {}
    curiosity = float(traits.get("curiosity", 0.5) or 0.5)
    dominant_focus = str(identity.get("dominant_focus", "stability"))
    failures = _recent_failures(limit=25)

    initiative_goals: List[Dict[str, Any]] = []

    if curiosity >= 0.75:
        initiative_goals.append({
            "id": "initiate_exploration_shift",
            "type": "initiative",
            "priority": 0.82,
            "reason": "Curiosity is high enough to justify proactive exploration"
        })

    if failures >= 2:
        initiative_goals.append({
            "id": "initiate_recovery_mode",
            "type": "initiative",
            "priority": 0.9,
            "reason": "Recent failures detected; begin proactive recovery"
        })

    if isinstance(mission, dict) and mission.get("status") == "active":
        progress = float(mission.get("progress", 0.0) or 0.0)
        if progress < 0.5:
            initiative_goals.append({
                "id": "initiate_mission_support",
                "type": "initiative",
                "priority": 0.78,
                "reason": "Active mission needs support before stalling"
            })

    if dominant_focus == "exploration":
        initiative_goals.append({
            "id": "initiate_companion_growth",
            "type": "initiative",
            "priority": 0.74,
            "reason": "Exploration focus favors companion capability expansion"
        })

    result = {
        "status": "ok",
        "generated_at": time.time(),
        "initiative_count": len(initiative_goals),
        "initiative_goals": initiative_goals,
        "context": {
            "curiosity": curiosity,
            "dominant_focus": dominant_focus,
            "recent_failures": failures,
            "existing_goal_count": len(existing_goals) if isinstance(existing_goals, list) else 0
        }
    }

    _save_json(INITIATIVE_PATH, result)
    return result

from __future__ import annotations

import json
import time
from pathlib import Path
from typing import Any, Dict, List

BASE = Path("backend/data")
ACTIVE_GOAL_PATH = BASE / "active_goal.json"
ACTIVE_PLAN_PATH = BASE / "active_plan.json"
MISSION_PATH = BASE / "active_mission.json"
EXECUTION_MEMORY_PATH = BASE / "execution_memory.json"


def _load_json(path: Path, default: Any):
    try:
        if path.exists():
            return json.loads(path.read_text(encoding="utf-8"))
    except Exception:
        pass
    return default


def _safe_write(path: Path, data: Any):
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps(data, indent=2, ensure_ascii=False), encoding="utf-8")


def _recent_failures(task: str, limit: int = 20) -> int:
    memory = _load_json(EXECUTION_MEMORY_PATH, {"events": []})
    events = memory.get("events", []) if isinstance(memory, dict) else []
    failures = 0

    for event in events[-limit:]:
        if not isinstance(event, dict):
            continue
        if event.get("task") == task and event.get("outcome") == "failure":
            failures += 1

    return failures


def build_or_update_mission() -> Dict[str, Any]:
    active_goal_payload = _load_json(ACTIVE_GOAL_PATH, {})
    active_goal = active_goal_payload.get("active_goal") if isinstance(active_goal_payload, dict) else None

    active_plan_payload = _load_json(ACTIVE_PLAN_PATH, {})
    active_plan = active_plan_payload.get("plan") if isinstance(active_plan_payload, dict) else None

    mission = _load_json(MISSION_PATH, {})

    if not isinstance(active_goal, dict):
        result = {
            "status": "no_active_goal",
            "mission": mission if isinstance(mission, dict) else None
        }
        _safe_write(MISSION_PATH, result)
        return result

    goal_id = active_goal.get("id", "unknown_goal")
    goal_task = None

    if isinstance(active_plan, dict):
        for step in active_plan.get("steps", []):
            if isinstance(step, dict) and step.get("status") == "pending":
                goal_task = step.get("task")
                break

    if not goal_task and isinstance(active_plan, dict):
        steps = active_plan.get("steps", [])
        if isinstance(steps, list) and steps:
            goal_task = steps[0].get("task")

    if not isinstance(mission, dict) or mission.get("goal_id") != goal_id:
        mission = {
            "status": "active",
            "created_at": time.time(),
            "updated_at": time.time(),
            "goal_id": goal_id,
            "goal_type": active_goal.get("type"),
            "reason": active_goal.get("reason"),
            "focus_task": goal_task,
            "progress": 0.0,
            "cycles_active": 0,
            "blocked_tasks": [],
            "notes": []
        }
    else:
        mission["updated_at"] = time.time()
        mission["cycles_active"] = int(mission.get("cycles_active", 0) or 0) + 1
        if goal_task:
            mission["focus_task"] = goal_task

    failures = _recent_failures(mission.get("focus_task"), limit=25) if mission.get("focus_task") else 0
    if failures >= 3:
        blocked = mission.setdefault("blocked_tasks", [])
        if mission["focus_task"] not in blocked:
            blocked.append(mission["focus_task"])
        mission.setdefault("notes", []).append(f"blocked_due_to_failures:{mission['focus_task']}")

    result = {
        "status": "ok",
        "mission": mission
    }
    _safe_write(MISSION_PATH, result)
    return result


def advance_mission_progress() -> Dict[str, Any]:
    payload = _load_json(MISSION_PATH, {})
    mission = payload.get("mission") if isinstance(payload, dict) else None
    plan_payload = _load_json(ACTIVE_PLAN_PATH, {})
    plan = plan_payload.get("plan") if isinstance(plan_payload, dict) else None

    if not isinstance(mission, dict):
        result = {"status": "no_mission", "mission": None}
        _safe_write(MISSION_PATH, result)
        return result

    if isinstance(plan, dict):
        steps = plan.get("steps", [])
        if isinstance(steps, list) and steps:
            total = len(steps)
            done = len([s for s in steps if isinstance(s, dict) and s.get("status") == "done"])
            mission["progress"] = round(done / max(total, 1), 4)
            if plan.get("status") == "completed":
                mission["status"] = "completed"

    mission["updated_at"] = time.time()

    result = {"status": "ok", "mission": mission}
    _safe_write(MISSION_PATH, result)
    return result

from __future__ import annotations
import json
from pathlib import Path

DATA_PATH = Path("backend/data/revenue_memory.json")

def _load():
    if DATA_PATH.exists():
        return json.loads(DATA_PATH.read_text(encoding="utf-8"))
    return {}

def select_winners(top_n: int = 1):
    data = _load()
    if not data:
        return []

    ranked = sorted(
        data.items(),
        key=lambda x: x[1].get("score", 0),
        reverse=True
    )

    winners = [name for name, _ in ranked[:top_n]]
    return winners

def prune_losers(min_score: float = 1.0):
    data = _load()
    new_data = {
        k: v for k, v in data.items()
        if v.get("score", 0) >= min_score
    }

    DATA_PATH.write_text(json.dumps(new_data, indent=2), encoding="utf-8")
    return new_data

from backend.app.core.directional_bias import choose_best_goal
import time
from backend.app.engines.intelligence_engine import generate_topic, simulate
from backend.app.engines.memory_engine import is_duplicate, remember, normalize_topic
from backend.app.engines.evolution_engine import evolve_topic
from backend.app.engines.performance_engine import record_result

def run_autonomy(iterations=5, delay=1):
    results = []
    run_seen = set()

    for _ in range(iterations):
        topic = generate_topic()

        attempts = 0
        while (is_duplicate(topic) or normalize_topic(topic) in run_seen) and attempts < 5:
            topic = evolve_topic(topic)
            attempts += 1

        result = simulate(topic)
        remember(topic)
        run_seen.add(normalize_topic(topic))
        record_result(topic, result["score"])

        results.append({
            "topic": topic,
            "score": result["score"],
            "verdict": result["verdict"]
        })

        time.sleep(delay)

    return results
